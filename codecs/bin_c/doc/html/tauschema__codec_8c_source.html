74 bool tausch_iter_is_ok( tausch_iter_t *iter )
75 {
76  bool notok = false;
77  notok |= iter->ebuf == NULL; // there is no buffer pointed
78  notok |= iter->idx == NULL; // idx is pointing to nothing
79  notok |= iter->idx >= iter->ebuf; // index is out of buffer or end of buffer
80  notok |= iter->next == NULL; // next is pointing to nothing
81  notok |= iter->next > iter->ebuf; // next is pointing out of buffer
82  notok |= (iter->next == iter->idx) && (iter->val != iter->next); // iter->val has incorrect value
83  return ! notok;
84 }

130 size_t tausch_decode_vluint( tausch_iter_t *iter )
131 {
132  size_t rv = 0;
133  size_t x = 0;
134  uint32_t s = 0;
135  if( iter->next != iter->val ) return ~0;
136  do
137  {
138  if( ! tausch_iter_is_ok( iter ) ) return ~0;
139  if( iter->next >= iter->ebuf )
140  {
141  iter->ebuf = NULL;
142  return ~0;
143  }
144  x = (size_t)(*iter->next);
145  if( (s+1) < (sizeof(size_t)*8) )
146  {
147  // we do not support bigger numbers for vluint
148  rv |= (x & 0x7f) << s;
149  s += 7;
150  }
151  iter->next += 1;
152  }
153  while( ((x & 0x80) == 0x80) && ( s < (sizeof(size_t)*8)) );
154  iter->val = iter->next;
155  return rv;
156 }

167 bool tausch_encode_vluint( tausch_iter_t *iter, size_t val )
168 {
169  size_t x = val;
170  uint8_t b = 0;
171  if( tausch_iter_is_complete(iter) ) return false; // the iterator is complete already
172  do
173  {
174  if( ! tausch_iter_is_ok( iter ) ) return false;
175  if( x & (~0x7f) )
176  b = 0x80;
177  else
178  b = 0x00;
179  b |= x & 0x7f;
180  *(iter->next++) = b;
181  x >>= 7;
182  }
183  while( x );
184  iter->val = iter->next;
185  return true;
186 }

195 size_t tausch_vluint_len( size_t val )
196 {
197  unsigned int *x = (unsigned int *)&val;
198  size_t rv = 0; // now: rv is number of msbits zeroed
199  for( int i = (sizeof(size_t)/sizeof(unsigned int)-1); i >= 0; i-- )
200  {
201  if( x[i] == 0 )
202  {
203  rv += 32;
204  }
205  else
206  {
207  rv += __builtin_clz( x[i] );
208  break;
209  }
210  }
211  rv = (sizeof(size_t)<<3) - rv; // now: rv is number of significant bits
212  rv = (rv+6) / 7; // now: rv is number of bytes
213  return rv;
214 }

223 bool tausch_decode_next( tausch_iter_t *iter )
224 {
225  size_t tag = 0;
226  size_t len = 0;
227  iter->idx = iter->next;
228  iter->val = iter->next;
229  while( 1 )
230  {
231  tag = tausch_decode_vluint( iter );
232  if( ! tausch_iter_is_ok( iter ) ) return false;
233  if( (tag & 3) == 3 )
234  {
235  // end of scope
236  if( (iter->scope == 0) || (tag > 3) )
237  {
238  // at the root level we stay at the END marker
239  // when tag part of the END is not 0, then it is also end of file
240  tausch_iter_init( iter, iter->idx, iter->ebuf - iter->idx );
241  return false;
242  }
243  iter->scope -= 1;
244  iter->idx = iter->next;
245  continue;
246  }
247  break;
248  }
249  len = 0;
250  if( (tag & 1) == 1 )
251  {
252  // collection or variadic tag
253  iter->scope += 1;
254  }
255  if( (tag & 2) == 2 )
256  {
257  len = tausch_decode_vluint( iter );
258  if( ! tausch_iter_is_ok( iter ) ) return false;
259  }
260  iter->lc = tag & 3;
261  iter->tag = tag >> 2;
262  iter->vlen = len;
263  if( len > 0 )
264  {
265  if( ((iter->next + len) <= iter->next) || ((iter->next + len) > iter->ebuf) )
266  {
267  iter->val = NULL;
268  iter->ebuf = NULL;
269  return false;
270  }
271  iter->val = iter->next;
272  iter->next += len; // idx may now equel to exbuf
273  }
274  else
275  {
276  iter->val = NULL;
277  }
278 
279  return true;
280 }

412 bool tausch_write_stuffing( tausch_iter_t *iter, size_t len )
413 {
414  // FIXME: check if the usage of this method matches what it does, like what happens with idx
415  if( ! tausch_iter_is_ok( iter ) ) return false;
416  size_t totlen = iter->next - iter->idx;
417  tausch_iter_t cpy = *iter; // copy the original iterator
418  bool is_eof = tausch_buf_is_eof( iter->idx );
419  if( ! tausch_iter_is_complete( iter ) )
420  {
421  // the iterator is incomplete
422  totlen =len;
423  }
424  bool finalize( bool rv )
425  {
426  iter->tag = 0;
427  if( !rv ) *iter = cpy;
428  if( is_eof && (iter->next < iter->ebuf) ) tausch_format_buf( iter->next );
429  return rv;
430  }
431  if( totlen < 1 ) return finalize( false );
432  if( ((iter->idx + totlen) <= iter->idx) || ((iter->idx + totlen) > iter->ebuf) ) return finalize( false );
433  iter->next = iter->idx;
434  iter->val = iter->next;
435  size_t tlv = 0;
436  if( totlen == 1 )
437  {
438  if( ! tausch_encode_vluint( iter, tlv ) ) return finalize( false );
439  iter->val = NULL;
440  }
441  else
442  {
443  tlv = 2;
444  if( ! tausch_encode_vluint( iter, tlv ) ) return finalize( false );
445  tlv = totlen - (size_t)iter->next + (size_t)iter->idx;
446  size_t lv = tausch_vluint_len( tlv ) , xv = tlv;
447  // find the amount of exact data
448  do { xv =tlv - lv; lv = tausch_vluint_len( xv ); } while( (xv+lv) != tlv );
449  if( ! tausch_encode_vluint( iter, xv ) ) return finalize( false );
450  iter->val = iter->next;
451  tlv = totlen - (size_t)iter->next + (size_t)iter->idx;
452  memset( iter->next, 0, tlv );
453  }
454  iter->next = iter->idx + totlen;
455  return finalize( true );
456 }

468 bool tausch_write_scope( tausch_iter_t *iter, size_t tag )
469 {
470  if( !tausch_iter_is_ok( iter ) ) return false;
471  if( tausch_iter_is_complete( iter ) ) return false; // scope can only be appended
472  if( iter->scope == ~(typeof(iter->scope))0 ) return false; // no more scopes can be added
473  size_t k = tag;
474  k <<= 2;
475  k += 1;
476  bool is_eof = tausch_buf_is_eof( iter->idx );
477  if( !tausch_encode_vluint( iter, k ) || (iter->next >= iter->ebuf) )
478  {
479  iter->next = iter->idx;
480  if( is_eof && (iter->next < iter->ebuf) ) tausch_format_buf( iter->next );
481  return false;
482  }
483  if( is_eof && (iter->next < iter->ebuf) ) tausch_format_buf( iter->next );
484  iter->val = NULL;
485  iter->lc = 1;
486  iter->scope += 1;
487  iter->tag = tag;
488  return true;
489 }

565 bool tausch_write_bool( tausch_iter_t *iter, size_t tag, bool *value )
566 {
567  if( ! tausch_iter_is_ok( iter ) ) return false;
568  bool val = true;
569  if( value != NULL ) val = *value;
570  bool is_eof = tausch_buf_is_eof( iter->idx );
571  tausch_iter_t cpy = *iter;
572  bool finalize( bool rv )
573  {
574  if( !rv ) *iter = cpy;
575  if( is_eof && (iter->next < iter->ebuf) ) tausch_format_buf( iter->next );
576  return rv;
577  }
578 
579  if( ! tausch_iter_is_complete( iter ) )
580  {
581  // Note if you want to append it in non space saving way, then
582  // before calling this method, do tausch_write_typX( iter, &tag, &value, 1 )
583  if( ! value )
584  {
585  // Append the element in a space saving way.
586  size_t tlv = tag << 2;
587  if( ! tausch_encode_vluint( iter, tlv ) ) return finalize( false );
588  iter->val = NULL;
589  }
590  else
591  {
592  // Write the item as uint8_t
593  if( ! tausch_write_typX( iter, tag, (uint8_t*)&val, sizeof(uint8_t))) return finalize( false );
594  }
595  iter->tag = tag;
596  return finalize( true );
597  }
598  else
599  {
600  // the iterator already has been read, we are overwriting
601  size_t totlen = iter->next - iter->idx;
602  tausch_iter_t tm = *iter;
603  tm.next = tm.idx;
604  tm.val = tm.next;
605  tm.ebuf = iter->next;
606  size_t bytes = tausch_vluint_len( tag << 2 );
607  if( totlen < bytes )
608  {
609  return finalize( false ); // can not encode, not enough space in it
610  }
611  else if( totlen <= (bytes+1) )
612  {
613  // only true value can be encoded
614  if( val == true )
615  {
616  size_t tlv = tag << 2;
617  if( ! tausch_encode_vluint( &tm, tlv+2 ) ) return finalize( false );
618  // append the single byte, length 0
619  if( totlen > bytes )
620  {
621  *(tm.next++) = 0x00;
622  }
623  iter->tag = tag;
624  }
625  else // val == false
626  {
627  if( ! tausch_write_stuffing( &tm, totlen ) ) return finalize( false );
628  iter->tag = 0;
629  }
630  return finalize( true );
631  }
632  else
633  {
634  // full TLV has to be written
635  size_t tlv = tag << 2;
636  tlv += 2; // encode the len too
637  if( ! tausch_encode_vluint( &tm, tlv ) ) return finalize( false );
638  tlv = totlen - bytes - 1; // now: tlv is length
639  if( ! tausch_encode_vluint( &tm, tlv ) ) return finalize( false );
640  while( tm.next < tm.ebuf )
641  {
642  *(tm.next++) = val;
643  val = 0;
644  }
645  iter->tag = tag;
646  return finalize( true );
647  }
648  }
649 
650  //return finalize( true );
651 }

687 size_t tausch_write_typX( tausch_iter_t *iter, size_t tag, uint8_t *value, size_t len )
688 {
689  if( ! tausch_iter_is_ok( iter ) ) return 0;
690  bool is_eof = tausch_buf_is_eof( iter->idx );
691  tausch_iter_t cpy = *iter;
692  size_t finalize( size_t rv )
693  {
694  if( !rv ) *iter = cpy;
695  if( is_eof && (iter->next < iter->ebuf) ) tausch_format_buf( iter->next );
696  return rv;
697  }
698  if( tausch_iter_is_complete( iter ) )
699  {
700  // we are overwriting the meomry
701  if( iter->tag != tag ) return finalize( 0 );
702  if( value != NULL )
703  {
704  // simple overwrite of the value
705  if( iter->vlen != len ) return finalize( 0 ); // the length must match
706  memcpy( iter->val, value, len );
707  return len;
708  }
709  else if( len > 0 )
710  {
711  // we are zeroing the value field
712  if( iter->vlen != len ) return finalize( 0 ); // lengths must match in this case
713  memset( iter->val, 0x00, len );
714  return len;
715  }
716  else
717  {
718  // we are turning the item to null in place
719  if( iter->vlen == 0 ) return finalize( 1 ); // it is already null
720  tag <<= 2;
721  // start new iterator for overwriting
722  tausch_iter_t tm = *iter;
723  tm.next = tm.idx;
724  tm.val = tm.next;
725  tm.ebuf = iter->next;
726  tm.lc = 0;
727  tm.vlen = 0;
728  if( ! tausch_encode_vluint( &tm, tag ) ) return finalize( 0 );
729  iter->next = tm.next;
730  iter->val = NULL;
731  iter->vlen = 0;
732  iter->lc = 0;
733  // fill the remainder with stuffing
734  size_t stlen = tm.ebuf - tm.next;
735  tm.idx = tm.next;
736  if( ! tausch_write_stuffing( &tm, stlen ) ) return finalize( 0 );
737  return finalize( 1 );
738  }
739  }
740  else if ( ! tausch_iter_is_clean( iter ) )
741  {
742  return finalize( 0 ); // if the iter is not clean we can not write
743  }
744  else
745  {
746  // we are appending to the stream
747  iter->tag = tag;
748  tag <<= 2;
749  if( len > 0 ) tag += 2;
750  iter->lc = tag & 3;
751  // write the tag
752  if( ! tausch_encode_vluint(iter, tag) ) return finalize( 0 );
753  if( len == 0 )
754  {
755  // null value was requested
756  iter->val = NULL;
757  iter->vlen = 0;
758  return finalize( 1 );
759  }
760  // write the len
761  if( ! tausch_encode_vluint(iter, len) ) return finalize( 0 );
762  // write the value
763  if( ((iter->next + len) <= iter->next) || ((iter->next + len) > iter->ebuf) ) return finalize( 0 ); // overflow
764  if( value != NULL )
765  {
766  memcpy( iter->next, value, len );
767  }
768  else
769  {
770  memset( iter->next, 0x00, len );
771  }
772  iter->next += len;
773  iter->vlen = len;
774  return finalize( len );
775  }
776 }
